generate one hot encoding

In [ ]:
a = np.random.randint(0,100,size=(10,10))
values = np.unique(a)
bit_count = len(values)
one_hot = np.zeros((10,10,bit_count))
for i in range(len(a)):
    for j in range(len(a[0])):
        index = np.where(values==a[i][j])[0][0]
        one_hot[i,j][index] = 1
print(one_hot)
